\copy liver_patient(
    Age,
    Gender,
    Total_Bilirubin,
    Direct_Bilirubin,
    Alkaline_Phosphotase,
    Alamine_Aminotransferase,
    Aspartate_Aminotransferase,
    Total_Protiens,
    Albumin,
    Albumin_and_Globulin_Ratio,
    Dataset
)
FROM 'INDIAN_LIVER_PATIENTS.csv'
WITH (FORMAT csv, HEADER, DELIMITER ',', QUOTE '"');
